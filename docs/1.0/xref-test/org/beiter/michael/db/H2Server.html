33  package org.beiter.michael.db;
34  
35  import org.h2.jdbcx.JdbcConnectionPool;
36  import org.h2.tools.Server;
37  import org.slf4j.Logger;
38  import org.slf4j.LoggerFactory;
39  
40  import java.sql.Connection;
41  import java.sql.ResultSet;
42  import java.sql.SQLException;
43  import java.util.concurrent.atomic.AtomicBoolean;
44  
45  /**
46   * A utility class that starts an H2 database server and populates it with a test table to ensure the connection works.
47   */
48  public class H2Server {
49  
50      private static final Logger LOG = LoggerFactory.getLogger(H2Server.class);
51  
52      public static final String DRIVER = "org.h2.Driver";
53      public static final String DATABASE = "h2test_db";
54      public static final String URL = "jdbc:h2:mem:" + DATABASE;
55      public static final String USER = "someUser";
56      public static final String PASSWORD = "noPassword";
57  
58      private static AtomicBoolean serverRunning = new AtomicBoolean(false);
59      private static volatile Server server;
60  
61      // prevent instantiation of this class
62      private H2Server() {
63      }
64  
65      /**
66       * Initializes the default schema and sets default values (if feasible). This deletes any data that are already in
67       * the database!
68       *
69       * @throws SQLException When there is a problem initializing the database. Note that this can leave the database in
70       *                      an inconsistent state.
71       */
72      public static synchronized void init()
73              throws SQLException {
74  
75          LOG.info("Initializing the in-memory database server with a default schema and default values");
76  
77          // connect to the DB (using the internal H2 connection pool) to trigger creation of an empty database owned by
78          // the correct user. The H2 documentation says that even if we throw the pool away immediately, using the pool
79          // is still faster than getting a direct connection with the driver manager...
80          JdbcConnectionPool cp = JdbcConnectionPool.create(URL, USER, PASSWORD);
81          Connection con = cp.getConnection();
82  
83          String stmt = "DROP TABLE IF EXISTS some_table";
84          con.prepareStatement(stmt).execute();
85  
86          // a user table that stores keys and values
87          stmt = "CREATE TABLE some_table ("
88                  + " id INT NOT NULL PRIMARY KEY,"
89                  + " key VARCHAR(255) NOT NULL,"
90                  + " value VARCHAR(255)"
91                  + ")";
92          con.prepareStatement(stmt).execute();
93  
94          stmt = "INSERT INTO some_table (id, key, value) VALUES "
95                  + "  (1, 'key1', 'value1')"
96                  + ", (2, 'key2', 'value2')"
97                  + ", (3, 'key3', 'value3')";
98          con.prepareStatement(stmt).execute();
99  
100         // check that the insert was successful
101         stmt = "SELECT COUNT(id) AS count FROM some_table";
102         ResultSet rs = con.prepareStatement(stmt).executeQuery();
103 
104         while (rs.next()) {
105             if (rs.getInt("count") != 3) {
106                 String error = "Result set row count mismatch: expected 3, found " + rs.getInt("count");
107                 LOG.warn(error);
108                 throw new IllegalStateException(error);
109             }
110         }
111 
112         con.close();
113         cp.dispose();
114     }
115 
116     /**
117      * Starts the server, but does not perform any database schema initialization
118      *
119      * @throws SQLException When the server cannot be started
120      */
121     public static void start()
122             throws SQLException {
123 
124         LOG.info("Checking if the in-memory database server can be started");
125 
126         // make sure the URL type and database type match
127         if (!URL.startsWith("jdbc:h2:mem:")) {
128             final String error = "Not an H2 connection URL: " + URL;
129             LOG.error(error);
130             throw new IllegalStateException(error);
131         }
132 
133         // load the driver
134         try {
135             Class.forName(DRIVER);
136         } catch (ClassNotFoundException e) {
137             // a SQL exception is not the ideal thing to throw here, but it is good enough for a testing setup like this
138             String error = "Driver class not found, cannot start the DB server: " + DRIVER;
139             LOG.error(error, e);
140             throw new SQLException(error, e);
141         }
142 
143         // start the server only if there is not already a server running
144         if (serverRunning.compareAndSet(false, true)) {
145 
146             try {
147                 LOG.info("Starting the in-memory database server");
148                 server = Server.createTcpServer();
149                 server.start();
150                 LOG.info("In-memory database server started");
151             } catch (SQLException e) {
152                 serverRunning.set(false);
153                 LOG.warn("In-memory database server not started dues to error");
154                 throw new SQLException(e);
155             }
156         } else {
157             LOG.info("In-memory database server already running");
158         }
159 
160 
161         // connect to the DB (using the internal H2 connection pool) to trigger creation of an empty database owned by
162         // the correct user. The H2 documentation says that even if we throw the pool away immediately, using the pool
163         // is still faster than getting a direct connection with the driver manager...
164         JdbcConnectionPool cp = JdbcConnectionPool.create(URL, USER, PASSWORD);
165         Connection conn = cp.getConnection();
166         conn.close();
167         cp.dispose();
168     }
169 
170     /**
171      * Stops the server
172      */
173     public static void stop() {
174 
175         LOG.info("Checking if the in-memory database server can be stopped");
176 
177         // stop the server only if there is a server running
178         if (serverRunning.compareAndSet(true, false)) {
179 
180             LOG.info("Stopping the in-memory database server");
181             server.stop();
182             LOG.info("In-memory database server stopped");
183         } else
184             LOG.info("In-memory database server not running");
185     }
186 
187     /**
188      * Restarts the server, but does not perform any database schema initialization
189      *
190      * @throws SQLException When the server cannot be started
191      */
192     public static void restart()
193             throws SQLException {
194 
195         stop();
196         start();
197     }
198 }
