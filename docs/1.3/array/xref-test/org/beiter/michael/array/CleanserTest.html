37  import java.security.SecureRandom;
38  import java.util.UUID;

40  import static org.hamcrest.Matchers.*;
41  import static org.junit.Assert.assertThat;
42  
43  public class CleanserTest {
44  
45      /**
46       * Test that the {@link Cleanser#wipe(byte[])} can handle {@code null} references
47       */
48      @Test
49      public void nullByteArrayTest() {
50  
51          byte[] empty = null;
52          Cleanser.wipe(empty);
53      }
54  
55      /**
56       * Test that the {@link Cleanser#wipe(byte[])} can handle empty source arrays
57       */
58      @Test
59      public void emptyByteArrayTest() {
60  
61          byte[] empty = new byte[0];
62          Cleanser.wipe(empty);
63      }
64  
65      /**
66       * Zero a byte array and assert that all values have been overwritten
67       */
68      @Test
69      public void zeroByteArrayTest() {
70  
71          byte[] random = new byte[16];
72          new SecureRandom().nextBytes(random);
73  
74          // Check that the array contains at least one byte that is not null. This makes the unit test probabilistic,
75          // but the probability of a secure random string with all zeros is extremely low.
76          int count = 0;
77          for (byte e : random)
78              if (e != (byte) 0)
79                  ++count;
80  
81          String error = "All elements in the random array are equal to 0";
82          assertThat(error, count, is(greaterThan(0)));
83  
84          // now zero the array, and make sure that all bytes are equal to 0
85          Cleanser.wipe(random);
86          count = 0;
87          for (byte e : random)
88              if (e != (byte) 0)
89                  ++count;
90  
91          error = "The random array contains at least one element that is unequal to 0";
92          assertThat(error, count, is(equalTo(0)));
93      }
94

115     /**
116      * Zero a char array and assert that all values have been overwritten
117      */
118     @Test
119     public void zeroCharArrayTest() {
120 
121         char[] random = UUID.randomUUID().toString().toCharArray();
122 
123         // Check that the array does not contain '\0' (works really well for UUIDs...)
124         int count = 0;
125         for (char e : random)
126             if (e == '\0')
127                 ++count;
128 
129         String error = "The random array contains at least one element that is equal to '\0'";
130         assertThat(error, count, is(equalTo(0)));
131 
132         // now zero the array, and make sure that all chars are equal to '\0'
133         Cleanser.wipe(random);
134         count = 0;
135         for (char e : random)
136             if (e != '\0')
137                 ++count;
138 
139         error = "The random array contains at least one element that is unequal to '\0'";
140         assertThat(error, count, is(equalTo(0)));
141     }
