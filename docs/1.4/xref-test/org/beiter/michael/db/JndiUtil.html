33  package org.beiter.michael.db;
34  
35  import org.h2.jdbcx.JdbcConnectionPool;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  import javax.naming.Context;
40  import javax.naming.InitialContext;
41  import javax.naming.NamingException;
42  
43  public class JndiUtil {
44  
45      private static final Logger LOG = LoggerFactory.getLogger(JndiUtil.class);
46  
47      private static boolean poolCreated = false;
48  
49      public static void setUpJndiConnectionPool(int maxConnections, String jndiName)
50              throws NamingException {
51  
52          if (poolCreated) {
53              LOG.info("Pool already exists, and JNDI name already registered");
54          } else {
55              LOG.info("Creating pool and registering JNDI name");
56  
57              // create a data source from the H2 default connection pool of the test database
58              JdbcConnectionPool pool = JdbcConnectionPool.create(H2Server.URL, H2Server.USER, H2Server.PASSWORD);
59              pool.setMaxConnections(maxConnections);
60              pool.setLoginTimeout(1); // set a short login timeout in case of errors
61  
62              // this requires the tomcat dependency to work, but is required to init and bind the context
63              System.setProperty(Context.INITIAL_CONTEXT_FACTORY, "org.apache.naming.java.javaURLContextFactory");
64              System.setProperty(Context.URL_PKG_PREFIXES, "org.apache.naming");
65  
66              // init and bind
67              Context context = new InitialContext();
68              context.createSubcontext("java:");
69              context.createSubcontext("java:/comp");
70              context.createSubcontext("java:/comp/env");
71              context.createSubcontext("java:/comp/env/jdbc");
72              context.bind(jndiName, pool);
73  
74              poolCreated = true;
75          }
76      }
77  }
