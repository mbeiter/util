33  package org.beiter.michael.db;
34  
35  /**
36   * This exception is thrown by Factories when they encounter an error that needs to be handled by the caller.
37   */
38  public class FactoryException
39          extends Exception {
40  
41      /**
42       * Serialization
43       */
44      private static final long serialVersionUID = 20141021L;
45  
46      /**
47       * @see Exception#Exception()
48       */
49      public FactoryException() {
50          super();
51      }
52  
53      /**
54       * @param message @see Exception#Exception(String, Throwable)
55       * @param cause   @see Exception#Exception(String, Throwable)
56       * @see Exception#Exception(String, Throwable)
57       */
58      public FactoryException(final String message, final Throwable cause) {
59          super(message, cause);
60      }
61  
62      /**
63       * @param message @see Exception#Exception(String)
64       * @see Exception#Exception(String)
65       */
66      public FactoryException(final String message) {
67          super(message);
68      }
69  
70      /**
71       * @param cause @see Exception#Exception(Throwable)
72       * @see Exception#Exception(Throwable)
73       */
74      public FactoryException(final Throwable cause) {
75          super(cause);
76      }
77  }
