733     /**
734      * The default TransactionIsolation state of connections created by this pool.
735      * <p>
736      * The following values are allowed:
737      * <ul>
738      * <li>Connection.TRANSACTION_NONE</li>
739      * <li>Connection.TRANSACTION_READ_COMMITTED</li>
740      * <li>Connection.TRANSACTION_READ_UNCOMMITTED</li>
741      * <li>Connection.TRANSACTION_REPEATABLE_READ</li>
742      * <li>Connection.TRANSACTION_SERIALIZABLE</li>
743      * </ul>
744      *
745      * @param defaultTransactionIsolation The default TransactionIsolation state of connections created by this pool
746      * @see java.sql.Connection
747      */
748     public final void setDefaultTransactionIsolation(final int defaultTransactionIsolation) {
749 
750         if (defaultTransactionIsolation != Connection.TRANSACTION_NONE
751                 && defaultTransactionIsolation != Connection.TRANSACTION_READ_COMMITTED
752                 && defaultTransactionIsolation != Connection.TRANSACTION_READ_UNCOMMITTED
